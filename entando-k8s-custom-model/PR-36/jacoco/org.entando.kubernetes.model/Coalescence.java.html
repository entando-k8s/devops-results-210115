package org.entando.kubernetes.model;

public class Coalescence {

    private Coalescence() {
    }

    @SuppressWarnings("unchecked")
    @SafeVarargs
    public static <T> T coalesce(T... t) {
        for (T item : t) {
            if (item != null) {
                return item;
            }
        }
        return null;
    }
}
