package org.entando.entando.web.common.model;

public enum FilterOperator {

    LIKE("like"),
    EQUAL("eq"),
    NOT_EQUAL("not"),
    GREATER("gt"),
    LOWER("lt");

    private final String value;

    private FilterOperator(String value) {
        this.value = value;
    }

    public String getValue() {
        return value;
    }

    public static FilterOperator parse(String op) {
        for (FilterOperator value : FilterOperator.values()) {
            if (value.getValue().equals(op)) {
                return value;
            }
        }
        throw new IllegalArgumentException(op + " is not a supported filter operator");
    }
}
